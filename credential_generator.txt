package credential;

import java.util.*;

class User 
{
    private String firstName;
    private String lastName;

    User(String firstName, String lastName) 
{
        this.firstName = firstName;
        this.lastName = lastName;
    }

    public String getFirstName() 
{
        return firstName;
    }
    public String getLastName() 
{
        return lastName;
    }
}

class CredentialService {
    private Random random = new Random();

        public String generateEmailAddress(String firstName, String lastName, String department) {
        return firstName.toLowerCase() + lastName.toLowerCase()  +".nisha"+ department +"@"+ "skit.ac.in";
    }

       public String generatePassword() {
        String capitalLetters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
        String smallLetters = "abcdefghijklmnopqrstuvwxyz";
        String numbers = "0123456789";
        String specialChars = "@#$%^&*!";

        String combined = capitalLetters + smallLetters + numbers + specialChars;

        StringBuilder password = new StringBuilder();

        password.append(capitalLetters.charAt(random.nextInt(capitalLetters.length())));
        password.append(smallLetters.charAt(random.nextInt(smallLetters.length())));
        password.append(numbers.charAt(random.nextInt(numbers.length())));
        password.append(specialChars.charAt(random.nextInt(specialChars.length())));

        for (int i = 4; i < 8; i++) 
        {
            password.append(combined.charAt(random.nextInt(combined.length())));
        }

        return password.toString();
    }

        public void showCredentials(User u, String email, String password) {
        System.out.println("\nDear " + u.getFirstName() + " your generated credentials are as follows:");
        System.out.println("Email ---->> " + email);
        System.out.println("Password ---->> " + password);
    }
}

public class CredentialGeneratorApp 
{
    public static void main(String[] args)
 {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter First Name: ");
        String firstName = sc.next();
        System.out.print("Enter Last Name: ");
        String lastName = sc.next();
        System.out.println("\nPlease enter the department from the following");
        System.out.println("1. Computer Science");
        System.out.println("2. Admin");
        System.out.println("3. Artificial Intelligence");
        System.out.println("4. Information Science");

        int choice = sc.nextInt();
        String department = "";
        switch (choice) 
      {
            case 1: department = "cse";
		    break;
            case 2: department = "admin"; 
		    break;
            case 3: department = "aiml"; 
		    break;
            case 4: department = "ise"; 
		    break;
            default: System.out.println("Invalid choice!"); return;
        }

        User u = new User(firstName, lastName);
        CredentialService cs = new CredentialService();
        String email = cs.generateEmailAddress(firstName, lastName, department);
        String password = cs.generatePassword();
        cs.showCredentials(u, email, password);
        sc.close();
    }
}
